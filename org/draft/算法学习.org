** 最大子序列
   求一个数组的最大子序列（和最大）
   #+BEGIN_SRC c
     int MaxSubseqSum(int A[], int N)
     {
       int ThisSum, MaxSum;
       int i;
       ThisSum = MaxSum = 0;
       for (i = 0; i < N; i++) {
         ThisSum += A[i];
         if (ThisSum > MaxSum)
           MaxSum = ThisSum;
         else
           ThisSum = 0;
       }
       return MaxSum;
     }
   #+END_SRC
